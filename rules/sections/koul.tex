\section*{\ar{كُل}}
\begin{enumerate}
\item \ar{كُل} + ουσιαστικό αόριστο = κάθε
	\begin{center}
	\begin{tabular}{ l r }
	κάθε σχολείο & \ar{كُل مَدرَسة} \\
	\end{tabular}
	\end{center}

\item \ar{كُل} + ουσιαστικό ορισμένο = όλο (σαν ενιαίο σύνολο)
	\begin{center}
	\begin{tabular}{ l r }
	όλο το σχολείο  & \ar{كُل المَدرَسة} \\
	όλο το απόγευμα & \ar{كُل المَساء}  \\
	\end{tabular}
	\end{center}
\end{enumerate}

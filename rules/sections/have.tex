\section*{Το ρήμα Έχω}

\begin{description}
\item[Άνθρωπος έχει πράγμα] κτητική αντωνυμία + \ar{ عِند } (στου - chez)

έχω ένα σπίτι

\item[Άνθρωπος έχει άνθρωπο] κτητική αντωνυμία + \ar{ لِـ } (για)

έχω έναν φίλο

\end{description}

\begin{enumerate}
\item Κλίνεται όπως το \ar{ كِتاب }
\item Αλλά αντί για \ar{ ـُ } βάζουμε \ar{ ـَ }
\end{enumerate}

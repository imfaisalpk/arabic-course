\section*{\ar{ لِـ } - \ar{ بِسَبَب } - \ar{ ﻵنَ }}

\begin{description}
\item [\ar{ بِسَبَب }] εξαιτίας + πάντα γεν κτητική
\begin{center}
\begin{tabular}{ c c }
εξαιτίας του καιρού & \ar{ بِسَبَب الجَو } \\
\end{tabular}
\end{center}

\item [\ar{ لِـ }]
	\begin{itemize}
	\item με ουσιασικά σημαίνει "για"
	\begin{center}
	\begin{tabular}{ c c }
	για ένα σπίτι & \ar{ لِبَيت }\textsuperscript{1} \\
	για το σπίτι  & \ar{ لِلبَيت } \\
	\end{tabular}

	\textsuperscript{1}\ar{ لِـ } + \ar{ الـ } $\rightarrow$ \ar{ لِلـ }
	\end{center}

	\item με ρήματα σημαίνει "για να"
	\begin{center}
	\begin{tabular}{ c c }
	για να πιείς & \ar{ لِتَشرَبُ } \\
	\end{tabular}
	\end{center}

	\end{itemize}

\end{description}

\backgroundsetup{
contents={%
% hack: backgroundsetup.position moves the picture too much. Ensure 
% there are enough vertical lines (100), and that each line is long 
% enough 
	\begin{tikzpicture}
		\foreach \fila in {0,...,100}
		{
			\draw [line width=1pt,color=notepadline] 
			(current page.west|-0,-\fila*\x*6) -- ++(\paperwidth*2,0);
		}
	\end{tikzpicture}%
},
	scale=1,
	angle=0,
	opacity=1,
	position={0,\x+2}
}
